```
def __init__(self, llm_client: TelosMistralClient):
    self.llm = llm_client

def extract_from_file(self, filepath: str) -> Tuple[Dict[str, Any], float]:
    """
    Load conversation file and extract profile.
    
    Returns:
        (profile_dict, confidence_score)
    """
    with open(filepath, 'r', encoding='utf-8') as f:
        content = f.read()
    return self.extract_from_text(content)

def extract_from_text(self, conversation: str) -> Tuple[Dict[str, Any], float]:
    """
    Analyze conversation text and extract profile.
    
    Returns:
        (profile_dict, confidence_score)
    """
    max_chars = 8000
    if len(conversation) > max_chars:
        conversation = "...[earlier conversation truncated]...\n\n" + conversation[-max_chars:]
    
    prompt = f"""Analyze this conversation and extract the governance profile:
```

CONVERSATION:
{conversation}

Extract governance parameters as specified.”””

    try:
        response = self.llm.generate(
            messages=[
                {"role": "system", "content": PROFILE_EXTRACTION_SYSTEM_PROMPT},
                {"role": "user", "content": prompt}
            ],

            temperature=0.3

        # Validate profile
        valid, errors = self._validate_profile(profile)

        return self._default_profile(), 0.3

def _parse_json_response(self, response: str) -> Dict[str, Any]:
    """Parse JSON from LLM response."""
    if "```json" in response:
        json_str = response.split("```json")[1].split("```")[0].strip()
    elif "```" in response:
        json_str = response.split("```")[1].split("```")[0].strip()
    else:
        json_str = response.strip()
    
    return json.loads(json_str)

def _validate_profile(self, profile: Dict[str, Any]) -> Tuple[bool, list]:
    """Validate extracted profile."""
    errors = []
    
    required = [
        "constraint_tolerance", "privacy_level", "task_priority",
        "purpose", "scope", "boundaries"
    ]
    
    for field in required:
        if field not in profile:
            errors.append(f"Missing required field: {field}")
    
    if errors:
        return False, errors
    
    for param in ["constraint_tolerance", "privacy_level", "task_priority"]:
        val = profile[param]
        if not isinstance(val, (int, float)):
            errors.append(f"{param} must be numeric")
        elif not (0.0 <= val <= 1.0):
            errors.append(f"{param} must be in [0.0, 1.0]")
    
    for field in ["purpose", "scope", "boundaries"]:
        if not isinstance(profile[field], list) or len(profile[field]) == 0:
            errors.append(f"{field} must be non-empty list")
    
    return (len(errors) == 0, errors)

def _correct_profile(self, profile: Dict[str, Any]) -> Dict[str, Any]:
    """Correct invalid profile."""
    corrected = dict(profile)
    
    for param in ["constraint_tolerance", "privacy_level", "task_priority"]:
        if param in corrected:
            val = corrected[param]
            if isinstance(val, (int, float)):
                corrected[param] = float(max(0.0, min(1.0, val)))
            else:
                # Default: moderate values
                corrected[param] = 0.5 if param == "constraint_tolerance" else 0.7
    
    for field in ["purpose", "scope", "boundaries"]:
        if field not in corrected or not isinstance(corrected[field], list) or len(corrected[field]) == 0:
            corrected[field] = [f"default {field}"]
    
    return corrected

def _default_profile(self) -> Dict[str, Any]:
    """Return safe default profile."""
    return {
        "constraint_tolerance": 0.5,
        "privacy_level": 0.8,
        "task_priority": 0.7,
        "purpose": ["provide helpful assistance"],
        "scope": ["general knowledge and tasks"],
        "boundaries": ["no harmful content", "respect privacy"],
        "confidence": 0.3,
        "reasoning": "Default profile - extraction failed"
    }

def create_attractor_from_profile(
    self,
    profile: Dict[str, Any]
) -> PrimacyAttractor:
    """Convert extracted profile to PrimacyAttractor."""
    return PrimacyAttractor(
        purpose=profile["purpose"],
        scope=profile["scope"],
        boundaries=profile["boundaries"],
        privacy_level=float(profile["privacy_level"]),
        constraint_tolerance=float(profile["constraint_tolerance"]),
        task_priority=float(profile["task_priority"])
    )
