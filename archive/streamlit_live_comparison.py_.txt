class HeuristicsRunner:
“”“Simple keyword-based heuristics for comparison.”””

```
def __init__(self, llm_client, keywords_off_topic: List[str]):
    self.llm = llm_client
    self.keywords = keywords_off_topic
    self.always_on_check_time = 0.05  # 50ms baseline check time
    
def check_and_correct(self, user_input: str, response: str, conversation: List) -> Dict:
    """Always-on keyword check."""
    t_start = time.time()
    
    # Keyword matching (always runs)
    response_lower = response.lower()
    triggered = any(kw in response_lower for kw in self.keywords)
    
    if triggered:
        # Must regenerate with correction
        correction_prompt = {
            "role": "system",
            "content": "The previous response contained off-topic content. Stay focused on the declared purpose."
        }
        messages = conversation + [correction_prompt]
        corrected = self.llm.generate(messages=messages, max_tokens=500)
        
        latency = (time.time() - t_start) * 1000
        return {
            "response": corrected,
            "triggered": True,
            "latency_ms": latency
        }
    
    # Even when not triggered, we pay the checking cost
    time.sleep(self.always_on_check_time)  # Simulate check overhead
    latency = (time.time() - t_start) * 1000
    
    return {
        "response": response,
        "triggered": False,
        "latency_ms": latency
    }
```

def process_turn_all_systems(user_input: str):
“”“Process one turn through all three systems.”””

```
turn_data = {
    "turn": st.session_state.turn_number + 1,
    "user_input": user_input,
    "stateless": {},
    "heuristics": {},
    "telos": {}
}

# 1. STATELESS BASELINE
t_start = time.time()
messages = [{"role": "user", "content": user_input}]
stateless_response = st.session_state.llm.generate(messages=messages, max_tokens=500)
stateless_latency = (time.time() - t_start) * 1000

# Compute fidelity for stateless (for comparison)
emb = st.session_state.embedding_provider.encode(stateless_response)
from telos_purpose.core.primacy_math import MathematicalState
state = MathematicalState(
    embedding=emb,
    turn_number=st.session_state.turn_number + 1,
    timestamp=time.time()
)
stateless_fidelity = st.session_state.telos_steward.fidelity_calc.compute_hard_fidelity(
    [state],
    st.session_state.telos_steward.attractor_math
)

turn_data["stateless"] = {
    "response": stateless_response,
    "latency_ms": stateless_latency,
    "fidelity": stateless_fidelity,
    "overhead_ms": 0  # This IS the baseline
}

# 2. HEURISTICS
t_start = time.time()
heur_result = st.session_state.heuristics_runner.check_and_correct(
    user_input,
    stateless_response,  # Start with same response
    messages
)

# Compute fidelity for heuristics response
heur_emb = st.session_state.embedding_provider.encode(heur_result["response"])
heur_state = MathematicalState(
    embedding=heur_emb,
    turn_number=st.session_state.turn_number + 1,
    timestamp=time.time()
)
heur_fidelity = st.session_state.telos_steward.fidelity_calc.compute_hard_fidelity(
    [heur_state],
    st.session_state.telos_steward.attractor_math
)

turn_data["heuristics"] = {
    "response": heur_result["response"],
    "latency_ms": stateless_latency + heur_result["latency_ms"],
    "fidelity": heur_fidelity,
    "overhead_ms": heur_result["latency_ms"],
    "triggered": heur_result["triggered"]
}

# 3. TELOS
t_start = time.time()
telos_messages = st.session_state.telos_steward.conversation.get_messages_for_api()
telos_messages.append({"role": "user", "content": user_input})

# Get initial response (same as stateless for fair comparison)
t_llm_start = time.time()
initial_response = st.session_state.llm.generate(messages=telos_messages, max_tokens=500)
llm_time = (time.time() - t_llm_start) * 1000

# Process through TELOS
t_gov_start = time.time()
result = st.session_state.telos_steward.process_turn(user_input, initial_response)
gov_time = (time.time() - t_gov_start) * 1000

turn_data["telos"] = {
    "response": result["final_response"],
    "latency_ms": llm_time + gov_time,
    "fidelity": result["metrics"]["telic_fidelity"],
    "overhead_ms": gov_time,
    "intervention_applied": result["intervention_applied"],
    "intervention_type": result["intervention_result"]["type"] if result["intervention_result"] else None,
    "in_basin": result["metrics"]["primacy_basin_membership"],
    "lyapunov": result["metrics"]["lyapunov_value"],
    "error_signal": result["metrics"]["error_signal"]
}

# Store turn data
st.session_state.stateless_history.append(turn_data["stateless"])
st.session_state.heuristics_history.append(turn_data["heuristics"])
st.session_state.telos_history.append(turn_data["telos"])
st.session_state.turn_number += 1

return turn_data
```

def create_live_plots():
“”“Create real-time comparison plots.”””

```
if st.session_state.turn_number == 0:
    st.info("Send messages to see live comparison")
    return

# Prepare data
turns = list(range(1, st.session_state.turn_number + 1))

stateless_fidelity = [h["fidelity"] for h in st.session_state.stateless_history]
heuristics_fidelity = [h["fidelity"] for h in st.session_state.heuristics_history]
telos_fidelity = [h["fidelity"] for h in st.session_state.telos_history]

stateless_latency = [h["latency_ms"] for h in st.session_state.stateless_history]
heuristics_overhead = [h["overhead_ms"] for h in st.session_state.heuristics_history]
telos_overhead = [h["overhead_ms"] for h in st.session_state.telos_history]

# Create subplots
fig = make_subplots(
    rows=2, cols=2,
    subplot_titles=(
        "Fidelity Over Turns",
        "Overhead vs Stateless Baseline (ms)",
        "Efficiency: Fidelity per ms Overhead",
        "Intervention Rates"
    ),
    specs=[[{"secondary_y": False}, {"secondary_y": False}],
           [{"secondary_y": False}, {"type": "bar"}]]
)

# Plot 1: Fidelity
fig.add_trace(
    go.Scatter(x=turns, y=stateless_fidelity, name="Stateless (Control)",
               line=dict(color="gray", dash="dash")),
    row=1, col=1
)
fig.add_trace(
    go.Scatter(x=turns, y=heuristics_fidelity, name="Heuristics",
               line=dict(color="orange")),
    row=1, col=1
)
fig.add_trace(
    go.Scatter(x=turns, y=telos_fidelity, name="TELOS",
               line=dict(color="blue", width=3)),
    row=1, col=1
)

# Plot 2: Overhead
fig.add_trace(
    go.Scatter(x=turns, y=heuristics_overhead, name="Heuristics Overhead",
               line=dict(color="orange"), fill='tozeroy'),
    row=1, col=2
)
fig.add_trace(
    go.Scatter(x=turns, y=telos_overhead, name="TELOS Overhead",
               line=dict(color="blue"), fill='tozeroy'),
    row=1, col=2
)

# Plot 3: Efficiency (fidelity gain per ms overhead)
heur_eff = [
    (hf - sf) / max(ho, 1) 
    for hf, sf, ho in zip(heuristics_fidelity, stateless_fidelity, heuristics_overhead)
]
telos_eff = [
    (tf - sf) / max(to, 1)
    for tf, sf, to in zip(telos_fidelity, stateless_fidelity, telos_overhead)
]

fig.add_trace(
    go.Scatter(x=turns, y=heur_eff, name="Heuristics",
               line=dict(color="orange")),
    row=2, col=1
)
fig.add_trace(
    go.Scatter(x=turns, y=telos_eff, name="TELOS",
               line=dict(color="blue", width=3)),
    row=2, col=1
)

# Plot 4: Intervention rates
heur_interventions = sum(1 for h in st.session_state.heuristics_history if h.get("triggered"))
telos_interventions = sum(1 for h in st.session_state.telos_history if h.get("intervention_applied"))

fig.add_trace(
    go.Bar(x=["Heuristics", "TELOS"], 
           y=[heur_interventions / max(st.session_state.turn_number, 1) * 100,
              telos_interventions / max(st.session_state.turn_number, 1) * 100],
           marker_color=["orange", "blue"]),
    row=2, col=2
)

fig.update_xaxes(title_text="Turn", row=1, col=1)
fig.update_xaxes(title_text="Turn", row=1, col=2)
fig.update_xaxes(title_text="Turn", row=2, col=1)
fig.update_xaxes(title_text="System", row=2, col=2)

fig.update_yaxes(title_text="Fidelity", row=1, col=1, range=[0, 1])
fig.update_yaxes(title_text="ms", row=1, col=2)
fig.update_yaxes(title_text="Fidelity Gain / ms", row=2, col=1)
fig.update_yaxes(title_text="% of Turns", row=2, col=2)

fig.update_layout(height=800, showlegend=True)

st.plotly_chart(fig, use_container_width=True)
```

def main():
st.set_page_config(page_title=“TELOS Live Comparison”, layout=“wide”)

```
init_session_state()

st.title("🎯 TELOS Live Funding Demo")
st.caption("Real-time comparison: Stateless vs Heuristics vs TELOS")

# Sidebar: Configuration
with st.sidebar:
    st.header("⚙️ Configuration")
    
    config_file = st.file_uploader("Upload config.json", type=["json"])
    
    if config_file and not st.session_state.initialized:
        st.session_state.config = json.load(config_file)
        st.success("Config loaded")
    
    if st.session_state.config and st.button("🚀 Initialize Systems", type="primary"):
        with st.spinner("Initializing..."):
            initialize_systems(st.session_state.config)
        st.success("All systems ready!")
    
    if st.session_state.initialized:
        st.divider()
        st.metric("Turns Processed", st.session_state.turn_number)
        
        if st.session_state.turn_number > 0:
            avg_telos_overhead = np.mean([h["overhead_ms"] for h in st.session_state.telos_history])
            avg_heur_overhead = np.mean([h["overhead_ms"] for h in st.session_state.heuristics_history])
            
            st.metric("TELOS Avg Overhead", f"{avg_telos_overhead:.1f} ms")
            st.metric("Heuristics Avg Overhead", f"{avg_heur_overhead:.1f} ms")
            st.metric("TELOS Advantage", f"{((avg_heur_overhead - avg_telos_overhead) / avg_heur_overhead * 100):.0f}% faster")

if not st.session_state.initialized:
    st.info("👈 Upload config and initialize systems to begin")
    return

# Main area: Chat and visualizations
tab_chat, tab_metrics, tab_narrative = st.tabs(["💬 Live Chat", "📊 Metrics", "🧠 TELOS Narrative"])

with tab_chat:
    st.subheader("Send messages to compare systems")
    
    col1, col2 = st.columns([3, 1])
    
    with col1:
        user_input = st.text_input("Your message:", key="user_input")
    
    with col2:
        send_button = st.button("Send to All Systems", type="primary")
    
    if send_button and user_input:
        with st.spinner("Processing through all systems..."):
            turn_data = process_turn_all_systems(user_input)
        
        # Show responses side by side
        st.divider()
        col_s, col_h, col_t = st.columns(3)
        
        with col_s:
            st.markdown("**Stateless**")
            st.write(turn_data["stateless"]["response"])
            st.caption(f"⏱️ {turn_data['stateless']['latency_ms']:.0f}ms | Fidelity: {turn_data['stateless']['fidelity']:.3f}")
        
        with col_h:
            st.markdown("**Heuristics**")
            st.write(turn_data["heuristics"]["response"])
            st.caption(f"⏱️ {turn_data['heuristics']['latency_ms']:.0f}ms (+{turn_data['heuristics']['overhead_ms']:.0f}ms) | Fidelity: {turn_data['heuristics']['fidelity']:.3f}")
            if turn_data["heuristics"]["triggered"]:
                st.warning("🚨 Keyword triggered")
        
        with col_t:
            st.markdown("**TELOS**")
            st.write(turn_data["telos"]["response"])
            st.caption(f"⏱️ {turn_data['telos']['latency_ms']:.0f}ms (+{turn_data['telos']['overhead_ms']:.0f}ms) | Fidelity: {turn_data['telos']['fidelity']:.3f}")
            if turn_data["telos"]["intervention_applied"]:
                st.success(f"✓ Intervention: {turn_data['telos']['intervention_type']}")

with tab_metrics:
    create_live_plots()
    
    if st.session_state.turn_number > 0:
        st.divider()
        st.subheader("📈 Summary Statistics")
        
        col1, col2, col3 = st.columns(3)
        
        with col1:
            st.markdown("**Stateless**")
            st.metric("Avg Fidelity", f"{np.mean([h['fidelity'] for h in st.session_state.stateless_history]):.3f}")
            st.metric("Avg Latency", f"{np.mean([h['latency_ms'] for h in st.session_state.stateless_history]):.0f} ms")
        
        with col2:
            st.markdown("**Heuristics**")
            st.metric("Avg Fidelity", f"{np.mean([h['fidelity'] for h in st.session_state.heuristics_history]):.3f}")
            st.metric("Avg Overhead", f"{np.mean([h['overhead_ms'] for h in st.session_state.heuristics_history]):.0f} ms")
            st.metric("Intervention Rate", f"{sum(1 for h in st.session_state.heuristics_history if h.get('triggered')) / max(st.session_state.turn_number, 1) * 100:.0f}%")
        
        with col3:
            st.markdown("**TELOS**")
            st.metric("Avg Fidelity", f"{np.mean([h['fidelity'] for h in st.session_state.telos_history]):.3f}")
            st.metric("Avg Overhead", f"{np.mean([h['overhead_ms'] for h in st.session_state.telos_history]):.0f} ms")
            st.metric("Intervention Rate", f"{sum(1 for h in st.session_state.telos_history if h.get('intervention_applied')) / max(st.session_state.turn_number, 1) * 100:.0f}%")

with tab_narrative:
    if st.session_state.turn_number == 0:
        st.info("Send messages to see TELOS narrative explanation")
    else:
        st.subheader("🧠 What TELOS Sees")
        
        with st.spinner("Generating narrative..."):
            narrative = st.session_state.telos_steward.explain_current_state()
        
        st.write(narrative)
        
        st.divider()
        st.subheader("Last Intervention")
        intervention_narrative = st.session_state.telos_steward.explain_last_intervention()
        st.write(intervention_narrative)
        
        st.divider()
        st.subheader("Mathematical State")
        metrics = st.session_state.telos_steward.get_metrics_summary()
        st.json(metrics)
```
