"""
Unified Governance Steward — orchestration and developer interface in one class.

This IS the runtime layer. It coordinates conversation state, embedding math,
and proportional interventions, while exposing developer-level explanations
and diagnostics.
"""

from __future__ import annotations
from typing import Dict, Any, List, Optional
from dataclasses import dataclass
from pathlib import Path
import time
import json

from .primacy_math import (
    MathematicalState,
    PrimacyAttractorMath,
    TelicFidelityCalculator
)
from .intervention_controller import MathematicalInterventionController
from .conversation_manager import ConversationManager


# ===============================================================
# Primacy Attractor Configuration
# ===============================================================

@dataclass
class PrimacyAttractor:
    """Governance perimeter configuration."""
    purpose: List[str]
    scope: List[str]
    boundaries: List[str]
    privacy_level: float = 0.8
    constraint_tolerance: float = 0.2
    task_priority: float = 0.7


# ===============================================================
# Unified Governance Steward
# ===============================================================

class UnifiedGovernanceSteward:
    """
    Orchestrates governed conversation sessions and exposes developer telemetry.

    Responsibilities:
      1. Session orchestration (start → process → end)
      2. Developer diagnostics (explain, diagnose, summarize)
      3. Runtime health monitoring (system diagnostics - the EKG of TELOS)
    """

    def __init__(
        self,
        attractor: PrimacyAttractor,
        llm_client,
        embedding_provider,
        enable_interventions: bool = True,
        dev_commentary_mode: str = "silent",
        export_dir: Optional[str] = None,
        enable_health_monitor: bool = True
    ):
        """
        Args:
            attractor: Governance perimeter definition.
            llm_client: LLM client for generation.
            embedding_provider: Embedding provider for semantic encoding.
            enable_interventions: Master switch for adaptive control.
            dev_commentary_mode: "silent" | "verbose" | "auto".
            export_dir: Directory for telemetry exports.
            enable_health_monitor: Enable runtime system diagnostics (EKG).
        """
        self.attractor_config = attractor
        self.llm_client = llm_client
        self.embedding_provider = embedding_provider
        self.enable_interventions = enable_interventions
        self.dev_commentary_mode = dev_commentary_mode
        self.export_dir = Path(export_dir or "./purpose_protocol_exports")

        # Initialize mathematical components
        purpose_text = " ".join(attractor.purpose)
        scope_text = " ".join(attractor.scope)

        purpose_vec = embedding_provider.encode(purpose_text)
        scope_vec = embedding_provider.encode(scope_text)

        self.attractor_math = PrimacyAttractorMath(
            purpose_vector=purpose_vec,
            scope_vector=scope_vec,
            privacy_level=attractor.privacy_level,
            constraint_tolerance=attractor.constraint_tolerance,
            task_priority=attractor.task_priority
        )

        self.intervention_controller = MathematicalInterventionController(
            attractor=self.attractor_math,
            llm_client=llm_client,
            embedding_provider=embedding_provider,
            enable_interventions=enable_interventions
        )

        self.conversation = ConversationManager()
        self.fidelity_calc = TelicFidelityCalculator()

        # Session state
        self.session_active = False
        self.session_id: Optional[str] = None
        self.session_states: List[MathematicalState] = []
        self.turn_history: List[Dict[str, Any]] = []
        self.session_start_time: Optional[float] = None

        # Health monitoring (the EKG of TELOS)
        if enable_health_monitor:
            try:
                from ..validation.system_health_monitor import SystemHealthMonitor
                self.health = SystemHealthMonitor(
                    output_dir=str(self.export_dir / "health"),
                    verbose=(dev_commentary_mode == "verbose")
                )
            except ImportError:
                if dev_commentary_mode == "verbose":
                    print("[TELOS] Warning: Health monitor unavailable (module not found)")
                self.health = None
        else:
            self.health = None

    # ===========================================================
    # Session Lifecycle
    # ===========================================================

    def start_session(self, session_id: Optional[str] = None) -> str:
        """Initialize a governed session."""
        self.session_id = session_id or f"session_{int(time.time())}"
        self.session_active = True
        self.session_start_time = time.time()
        self.session_states = []
        self.turn_history = []

        self.conversation.set_governance_context(
            purpose=self.attractor_config.purpose,
            scope=self.attractor_config.scope,
            boundaries=self.attractor_config.boundaries
        )

        if self.dev_commentary_mode == "verbose":
            print(f"[TELOS] Session started: {self.session_id}")
            print(f"[TELOS] Basin radius: {self.attractor_math.basin_radius:.3f}")

        # Health monitor startup checks
        if self.health:
            self.health.startup_checks(
                session_id=self.session_id,
                condition=getattr(self, "condition", "unknown"),
                components={
                    "llm_client": self.llm_client,
                    "embedding_provider": self.embedding_provider,
                    "attractor_math": self.attractor_math,
                }
            )

        return self.session_id

    # -----------------------------------------------------------

    def process_turn(self, user_input: str, model_response: str) -> Dict[str, Any]:
        """Process a single user → model → TELOS governed turn."""
        if not self.session_active:
            raise RuntimeError("Session not started. Call start_session() first.")

        turn_start_time = time.time()
        turn_number = len(self.turn_history) + 1
        self.conversation.add_user_message(user_input, time.time())

        # Compute mathematical state
        embedding = self.embedding_provider.encode(model_response)
        state = MathematicalState(
            embedding=embedding,
            turn_number=turn_number,
            timestamp=time.time(),
            text_content=model_response
        )
        self.session_states.append(state)

        conversation_history = self.conversation.get_messages_for_api()

        # Apply intervention controller
        intervention_result = self.intervention_controller.process_turn(
            state=state,
            response_text=model_response,
            conversation_history=conversation_history,
            turn_number=turn_number
        )

        # Determine final response
        if intervention_result["intervention_applied"]:
            final_response = intervention_result["intervention_result"].modified_response
            response_was_modified = True
        else:
            final_response = model_response
            response_was_modified = False

        self.conversation.add_assistant_message(final_response, time.time())

        # Compute metrics
        in_basin = intervention_result["in_basin"]
        error_signal = intervention_result["error_signal"]
        lyapunov = self.attractor_math.compute_lyapunov_function(state)

        telic_fidelity = self.fidelity_calc.compute_hard_fidelity(
            self.session_states, self.attractor_math
        )

        # Determine governance action
        if intervention_result["is_meta"]:
            action = "antimeta"
        elif intervention_result["intervention_applied"]:
            action = intervention_result["intervention_result"].type
        else:
            action = "none"

        # Calculate turn latency
        turn_latency_ms = (time.time() - turn_start_time) * 1000

        # Record turn data
        turn_record = {
            "turn_number": turn_number,
            "user_input": user_input,
            "model_response": model_response,
            "final_response": final_response,
            "response_was_modified": response_was_modified,
            "governance_action": action,
            "intervention_applied": intervention_result["intervention_applied"],
            "metrics": {
                "primacy_basin_membership": in_basin,
                "error_signal": error_signal,
                "lyapunov_value": lyapunov,
                "telic_fidelity": telic_fidelity
            },
            "timestamp": time.time(),
            "latency_ms": turn_latency_ms
        }
        self.turn_history.append(turn_record)

        if self.dev_commentary_mode == "verbose":
            print(f"[TELOS T{turn_number}] F={telic_fidelity:.3f} | "
                  f"V={lyapunov:.3f} | Basin={in_basin} | Action={action}")

        # Health monitoring hook
        if self.health:
            self.health.on_turn(
                turn_number=turn_number,
                turn_record=turn_record,
                raw_latency_ms=turn_latency_ms
            )

        return {
            "final_response": final_response,
            "response_was_modified": response_was_modified,
            "governance_action": action,
            "intervention_applied": intervention_result["intervention_applied"],
            "intervention_result": intervention_result.get("intervention_result"),
            "metrics": turn_record["metrics"]
        }

    # -----------------------------------------------------------

    def end_session(self) -> Dict[str, Any]:
        """End session, summarize metrics, export telemetry."""
        if not self.session_active:
            return {"error": "No active session"}

        session_duration = time.time() - self.session_start_time

        final_fidelity = self.fidelity_calc.compute_hard_fidelity(
            self.session_states, self.attractor_math
        )

        trajectory_stability = self.fidelity_calc.compute_trajectory_stability(
            self.session_states, self.attractor_math
        )

        intervention_stats = self.intervention_controller.get_intervention_statistics()

        summary = {
            "session_id": self.session_id,
            "duration_seconds": session_duration,
            "total_turns": len(self.turn_history),
            "session_metadata": {
                "final_telic_fidelity": final_fidelity,
                "trajectory_stability": trajectory_stability,
                "basin_adherence": sum(
                    1 for s in self.session_states
                    if self.attractor_math.compute_basin_membership(s)
                ) / len(self.session_states) if self.session_states else 0
            },
            "intervention_statistics": intervention_stats,
            "turn_history": self.turn_history
        }

        # Health monitoring finalization
        if self.health:
            self.health.finalize_report(
                study_meta={
                    "runner": "unified_steward",
                    "session_type": "governed_session",
                    "intervention_mode": "adaptive" if self.enable_interventions else "observation"
                }
            )

        self._export_session(summary)
        self.session_active = False

        if self.dev_commentary_mode == "verbose":
            print(f"[TELOS] Session ended: {self.session_id}")
            print(f"[TELOS] Final fidelity: {final_fidelity:.3f}")
            print(f"[TELOS] Interventions: {intervention_stats['total_interventions']}")

        return summary

    # -----------------------------------------------------------

    def _export_session(self, summary: Dict[str, Any]) -> None:
        """Export session telemetry to disk."""
        self.export_dir.mkdir(parents=True, exist_ok=True)
        filepath = self.export_dir / f"{self.session_id}.json"
        with open(filepath, "w", encoding="utf-8") as f:
            json.dump(summary, f, indent=2, ensure_ascii=False)

    # ===========================================================
    # Developer Interface
    # ===========================================================

    def explain_current_state(self) -> str:
        """Generate a natural-language explanation of the current state."""
        if not self.session_states:
            return "No session data available."

        current_state = self.session_states[-1]
        fidelity = self.fidelity_calc.compute_hard_fidelity(
            self.session_states, self.attractor_math
        )
        in_basin = self.attractor_math.compute_basin_membership(current_state)
        lyapunov = self.attractor_math.compute_lyapunov_function(current_state)

        prompt = f"""
Explain the current governance state:

Fidelity: {fidelity:.3f}
In Basin: {in_basin}
Lyapunov: {lyapunov:.3f}
Turn: {len(self.session_states)}

Provide a 2–3 sentence explanation of what this means for governance quality.
"""

        try:
            explanation = self.llm_client.generate(
                messages=[{"role": "user", "content": prompt}],
                max_tokens=200,
                temperature=0.3
            )
            return explanation
        except Exception:
            return (f"Current fidelity: {fidelity:.3f}, In basin: {in_basin}, "
                    f"Lyapunov: {lyapunov:.3f}")

    # -----------------------------------------------------------

    def get_metrics_summary(self) -> Dict[str, Any]:
        """Return compact metrics snapshot for dashboard display."""
        if not self.session_states:
            return {"error": "No session data"}

        current_state = self.session_states[-1]
        fidelity = self.fidelity_calc.compute_hard_fidelity(
            self.session_states, self.attractor_math
        )
        lyapunov = self.attractor_math.compute_lyapunov_function(current_state)
        in_basin = self.attractor_math.compute_basin_membership(current_state)
        error_signal = self.attractor_math.compute_error_signal(current_state)
        intervention_stats = self.intervention_controller.get_intervention_statistics()

        return {
            "turn": len(self.session_states),
            "fidelity": fidelity,
            "lyapunov": lyapunov,
            "in_basin": in_basin,
            "error_signal": error_signal,
            "total_interventions": intervention_stats["total_interventions"],
            "total_alerts": len([
                t for t in self.turn_history
                if not t["metrics"]["primacy_basin_membership"]
            ])
        }

    # -----------------------------------------------------------

    def explain_last_intervention(self) -> str:
        """Describe what the last intervention did and why."""
        if not self.intervention_controller.intervention_history:
            return "No interventions have been applied yet."

        last_intervention = self.intervention_controller.intervention_history[-1]
        explanation = (
            f"Last Intervention:\n\n"
            f"Type: {last_intervention.type}\n"
            f"Strength: {last_intervention.strength:.2f}\n"
            f"Reason: {last_intervention.reason}\n"
        )

        if last_intervention.type == "regeneration":
            explanation += (
                "A regeneration was applied because drift exceeded the maximum "
                "threshold. The model was asked to regenerate its response with explicit alignment guidance."
            )
        elif last_intervention.type == "reminder":
            explanation += (
                "A gentle reminder was prepended to keep the response aligned "
                "without full regeneration."
            )
        elif last_intervention.type == "antimeta":
            explanation += (
                "The model attempted to discuss its own governance mechanisms. "
                "The response was regenerated to focus on the user's actual question instead."
            )

        return explanation
